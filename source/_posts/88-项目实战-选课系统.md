class Teacher{
    constructor(){
        this.courses=[]
    }
    openCourse(course){
        this.courses.push(course);
    }
    
    closeCourse(course){
        for(let c of this.courses){
            if(course.id===c.id){
                this.courses.remove(course)
            }
        }       
    }
    
}


class Course{
    constructor(id,name,startTime,addr){
        this.id=id
        this.name=name
        this.startTime=startTime
        this.addr=addr
        this.students=[]
        this.teachers=[]
    }
}

class Student{
    constructor(){
        this.courses=[]
    }
    
    selectCourse(course){
        if(course.teachers){
            this.courses.push(course)

    }
    
    cancleCourse(course){
        for(let c of this.courses){
            if(course.id===c.id){
                this.courses.remove(course)
            }
        }
    }
}
